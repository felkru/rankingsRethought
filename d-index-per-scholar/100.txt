    <h1>
    Best Universities in the World 2023 Ranking
            </h1>

<div>
    <div class="ads-margin ads-margin--top-section hide-tablet">
            </div>
    <p data-gs="1620">
    The 2nd edition of Research.com ranking of the best universities in the world is
    based on metrics gathered from Microsoft Academic Graph on
    21-12-2022. Position in our world university ranking is based on total H-index
    values of all leading scientists affiliated with a specific
    university. A leading scientist is a scholar with a D-index (discipline H-index) above a
    predetermined threshold (typically 30 or 40) for their
    scientific disciplines. Our ranking of top
    universities in the world only lists 1000 universities with the highest H-index total.
    <span class="show-more" onclick="showMoreText(event)" style="display: none;">Show more</span>
</p>

<div class="show-more-text" style="display: block;">
    <p>
        The ranking process involved a careful examination of
        166,880 researchers on Google Scholar and Microsoft Academic Graph.
    </p>
    <p>
        A detailed definition of our research process
        can be examined on our
        <a href="/our-methodology" title="Our Methodology">
            methodology page
        </a>.
    </p>
    <p>
        Please be advised that the research institution or university for a
        scientist is typically chosen based on the affiliation details
        available on their Google Scholar profile.
    </p>
    <p>
        In comparison to other global university rankings our aim is to focus
        mostly on the scientific accomplishments of
        leading scientists associated with each institution. We
        wish to inspire scientists, business-owners, and
        institutions worldwide to
        learn where current experts are heading and to
        provide an opportunity for the entire scientific community to
        confirm who the leading experts in specific
        fields of research, in particular geographies, or
        even within research organizations are.
    </p>
</div>

<script>
    const form = document.getElementById('newsletter-form')

    if (form) {

        const formParent = document.getElementById('newsletter')
        const formConfirmation = document.getElementById('newsletter-confirmation')
        const formError = document.getElementById('newsletter-error')
        const submitButton = document.getElementById('newsletter-submit')

        form.querySelector('input').addEventListener('input', () => {
            if (formError !== null) {
                formError.classList.add('hidden');
            }
        });

        form.addEventListener('submit', async (event) => {
            event.preventDefault();
            const email = document.getElementById("newsletter-email").value;
            if (!testEmail(email)) {
                if (formError !== null) {
                    formError.innerHTML = 'Please enter a valid email address';
                    formError.classList.remove('hidden');
                }
                return;
            }

            formError.classList.add('hidden');
            setFormBusy()

            try {
                const response = await fetch(form.getAttribute('action'), {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Accept': 'application/json',
                        'X-CSRF-TOKEN': document.querySelector('meta[name="csrf-token"]').getAttribute('content'),
                    },
                    body: JSON.stringify({
                        email: email,
                    }),
                });

                if (response.ok) {
                    formParent.classList.add('hidden');
                    formConfirmation.classList.remove('hidden');
                    return;
                }

                if (response.status === 422) {
                    const responseData = await response.json();
                    const errorMessage = responseData?.message ?? responseData?.error
                    showEmailInputError(errorMessage);
                    return
                }

                showEmailInputError();
            } catch {
                showEmailInputError();
            } finally {
                submitButton.classList.remove('disabled');
                setFormNotBusy()
            }
        });

        function showEmailInputError(message = 'An error occurred. Please try again later.') {
            setFormNotBusy();
            formError.innerHTML = message;
            formError.classList.remove('hidden');
        }

        function testEmail(email) {
            const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
            return emailRegex.test(email);
        }

        function setFormBusy() {
            submitButton.classList.add('disabled');
        }

        function setFormNotBusy() {
            submitButton.classList.remove('disabled');
        }
    }


</script>
